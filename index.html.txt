<script>
const quiz = [
  {
    question: "취재를 시작할 때 당신은?",
    answers: [
      { text: "자료부터 철저히 조사한다", type: "A" },
      { text: "현장으로 바로 간다", type: "B" }
    ]
  },
  {
    question: "인터뷰 스타일은?",
    answers: [
      { text: "논리적으로 차근차근 질문", type: "A" },
      { text: "분위기를 풀면서 자연스럽게", type: "B" }
    ]
  },
  {
    question: "기사 작성 스타일은?",
    answers: [
      { text: "사실 위주 정확한 기사", type: "A" },
      { text: "스토리텔링 중심 기사", type: "B" }
    ]
  }
];

let current = 0;
let score = { A: 0, B: 0 };

const questionBox = document.getElementById("questionBox");
const answersBox = document.getElementById("answers");
const startBtn = document.getElementById("startBtn");

startBtn.onclick = () => {
  startBtn.classList.add("hidden");
  showQuestion();
};

function showQuestion() {
  const q = quiz[current];
  questionBox.innerHTML = `<h2>${q.question}</h2>`;
  answersBox.innerHTML = "";

  q.answers.forEach(ans => {
    const btn = document.createElement("button");
    btn.innerText = ans.text;
    btn.onclick = () => selectAnswer(ans.type);
    answersBox.appendChild(btn);
  });
}

function selectAnswer(type) {
  score[type]++;
  current++;

  if (current < quiz.length) {
    showQuestion();
  } else {
    showResult();
  }
}

function showResult() {
  questionBox.innerHTML = "";
  answersBox.innerHTML = "";

  let result = score.A > score.B ? "분석형 기자 🧠" : "현장형 기자 🎤";

  questionBox.innerHTML = `<h2>당신의 결과는?</h2>`;
  answersBox.innerHTML = `<h3>${result}</h3>
  <p>${result === "분석형 기자 🧠"
    ? "철저한 분석과 논리로 기사를 작성하는 타입!"
    : "현장에서 생생한 이야기를 전달하는 타입!"}
  </p>
  <button onclick="location.reload()">다시하기</button>`;
}
</script>
